void leernum(int &n);
void proceso (int n,int &j);
void reporte (int j);

int main(int argc, char *argv[]) {
	int n,j;
	leernum (n);
	proceso (n,j);
	reporte (j);
	return 0;
}
void leernum(int &n){
	do{
		cout<<"ingrese un numero: ";
		cin>>n;
	} while(n<=0);
}
void proceso (int n,int &j){
	int i;
	for (i=n;i>=1;i--){
		cout<<endl;
		for (j=1;j<=i;j++)
			cout<<"*";
	}
}
void reporte (int j){
	cout<<j<<" * "<<endl;
}
